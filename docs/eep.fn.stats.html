Get the sample variance of all the things

 
function VarianceFunction() {
  var self = this, m = 0, m2 = 0, d = 0, n = 0;
  self.name = "vars";
  self.init = function() { };
  self.accumulate = function(v) { n+=1; d = v - m; m = m + d/n; m2 = m2 + d*(v - m); };
  self.emit = function()  { return m2/(n-1); };
  self.make = function() { return new VarianceFunction(); };
}

Get the standard deviation of all the things

 
function StdevFunction() {
  var self = this, m, m2, d, n;
  self.name = "stdevs";
  self.init = function() { m = 0, m2 = 0, d = 0, n = 0; };
  self.accumulate = function(v) {
    n+=1; 
    d = v - m;
    m = m + d/n;
    m2 = m2 + d*(v-m); };
  self.emit = function()  { return Math.sqrt(m2/(n-1)); };
  self.make = function() { return new StdevFunction(); };
}

Get the kurtosis of all the things

 
function KurtosisFunction() {
  var self = this, m = 0, m4 = 0, m3 = 0, m2 = 0, d = 0, n = 0;
  self.name = "kurtosis";
  self.init = function() { };
  self.accumulate = function(v) { 
    var o = n;
    n+=1; 
    d = v - m; 
    var dn = d / n, dn2 = dn * dn, t1 = d * dn *  o;
    m = m + dn;
    m4 = m4 + t1 * dn2 * (n*n - 3*n + 3) + 6 * dn2 * m2 - 4 * dn * m3;
    m3 = m3 + t1 * dn * (n - 2) - 3 * dn * m2;
    m2 = m2 + t1;
  };
  self.emit = function()  { return (n*m4)/(m2*m2) - 3; };
  self.make = function() { return new KurtosisFunction(); };
}
util.inherits(KurtosisFunction, eep.AggregateFunction);

function AllStatsFunction() {
  var self = this, m, m4, m3, m3, d, n, s, min, max;
  self.name = "kurtosis";
  self.init = function() { 
    m = 0, m4 = 0, m3 = 0, m2 = 0, d = 0, n = 0, s = 0, min = null, max = null;
  };
  self.accumulate = function(v) {
    if (v == null) return;
    var o = n;
    s+=v;
    if (v != null) { if (min == null) { min = v; } else { min = (v < min) ? v : min; } };
    if (v != null) { if (max == null) { max = v; } else { max = (v > max) ? v : max; } };
    n+=1;
    d = v - m;
    var dn = d / n, dn2 = dn * dn, t1 = d * dn *  o;
    m = m + dn;
    m4 = m4 + t1 * dn2 * (n*n - 3*n + 3) + 6 * dn2 * m2 - 4 * dn * m3;
    m3 = m3 + t1 * dn * (n - 2) - 3 * dn * m2;
    m2 = m2 + t1;
  };
  self.emit = function()  { 
    var variance = m2/(n-1);
    return results = {
      count: n,
      sum: s,
      min: min,
      max: max,
      mean: m,
      vars: variance,
      stdevs:  Math.sqrt(variance),
      kurtosis: (n*m4)/(m2*m2) - 3 
    };
  };
  self.make = function() { return new AllStatsFunction(); };
}
util.inherits(AllStatsFunction, eep.AggregateFunction);
